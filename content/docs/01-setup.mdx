import { Step, Steps } from "fumadocs-ui/components/steps";

<Steps>
  <Step>
    Clone workshop template from v0 template into your personal v0

    https://v0.app/templates/ai-ship-workshop-tyVvKQthD7X

    > **Note:**  
    > Ensure that Vercel integrations are properly connected within v0. This is essential for synchronizing environment variables, which will be required for local development and integrations such as:
    > - AI Gateway
    > - Redis

  </Step>
  <Step>
    Deploy project from v0 to Vercel and verify deployment from v0
  </Step>

  <Step>
    Clone GitHub repository to your local machine
  </Step>
  <Step>
    Install dependencies
    ```bash
    pnpm install
    ```
  </Step>
  <Step>
    Link your local project to vercel project
    ```bash
    vercel link
    ```
  </Step>
  <Step>
    Set up environment variables
    ```bash
    vercel env pull
    ```
  </Step>
</Steps>
