<Steps>
  <Step>
    Clone workshop template from v0 template into your personal v0

    https://v0.app/templates/ai-ship-workshop-tyVvKQthD7X

  </Step>
  <Step>
    Deploy project from v0 to Vercel and verify deployment from v0
  </Step>

  <Step>
    Clone GitHub repository to your local machine
  </Step>
  <Step>
    Run `pnpm install` to install dependencies
  </Step>
  <Step>
    Run `vercel link` to link your local project to vercel project
  </Step>
  <Step>
    Set up environment variables with `vercel env pull`
  </Step>
</Steps>
